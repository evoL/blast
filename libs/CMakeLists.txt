# SDL
set (
  SDL_FOLDER ${CMAKE_CURRENT_SOURCE_DIR}/sdl-1.3
  CACHE STRING "Path to SDL 1.3"
)

find_library (SDL_LIBRARY
  NAMES SDL-1.3.0 SDL
  PATHS ${SDL_FOLDER}
  PATH_SUFFIXES win32 darwin
  NO_DEFAULT_PATH
)
find_library (SDLMAIN_LIBRARY
  NAMES SDLmain
  PATHS ${SDL_FOLDER}
  PATH_SUFFIXES win32 darwin
  NO_DEFAULT_PATH
)
